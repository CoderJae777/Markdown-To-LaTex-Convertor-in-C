\subsection{ 16. Known Limitations and Unsupported Features}

\subsubsection{ 16.1 Unsupported Markdown Syntax}

\begin{itemize}
\item  Horizontal rules (----, ****, \_\_\_\_)
\item  Strikethrough (\~{}\~{}text\~{}\~{})
\item  Subscript and superscript (H\~{}2\~{}O, x\^{}2\^{})
\item  Task lists (- [ ] Task, - [x] Done)
\item  Footnotes ([\^{}1], [\^{}1]: Note text)
\item  Definition lists (Term : Definition)
\item  HTML blocks (<div>, <table> tags)
\item  Raw HTML (<span style="color:red">text</span>)
\end{itemize}

\subsubsection{ 16.2 Limited or Partial Support}

\begin{itemize}
\item  Escaped characters (may not work as expected)
\item  Nested formatting edge cases
\item  Very deep list nesting (may have depth limits)
\item  Unicode and emoji support
\end{itemize}

\begin{itemize}
\item  Multiple consecutive blank lines
